// categoria = livro AND
// (tags = imutabilidade OR tags = larga escala)
// AND NOT(nome = scala)
GET /produtos/_search
{
  "query" : {
    "bool": {
      "must": {"match": {"categoria": "livro" }},
      "should": [
        {"match": {"tags": "imutabilidade" }},
        {"match": {"tags": "larga escala" }}
      ],
      "must_not": {"match": {"nome": "scala"}}
    }
  }
}
